function farmStats(item, fid){
  const counts=[...(item.counts?.[fid]||[])].sort((a,b)=>a.date.localeCompare(b.date));
  const last=counts[counts.length-1]||null, prev=counts[counts.length-2]||null;
  const lastKg=last?countToKg(item,last.qtys):null;
  const fd=item.feedDaily?.[fid];
  let baseDaily=null, source=null;
  if(fd&&fd.dailyKg>0){ baseDaily=fd.dailyKg; source="excel"; }
  else if(last&&prev){
    const days=Math.max(1,Math.round((new Date(last.date)-new Date(prev.date))/86400000));
    const diff=countToKg(item,prev.qtys)-lastKg;
    baseDaily=diff>0?diff/days:0; source="count";
  }
  const extra=(item.extraDaily?.[fid])||0;   // 別給与（1日あたりkg・Excel/棚卸しに含まれない分）
  let dailyKg=baseDaily!=null?baseDaily+extra:(extra>0?extra:null);
  if(baseDaily==null&&extra>0) source="extra";
  const usage=dailyKg!=null?dailyKg*30:null;
  return { last,lastKg,usage,dailyKg,baseDaily,extra,source,lastDate:last?.date||null };
}
function analyze(item){
  let totalStock=0,totalUsage=0,hasUsage=false,hasExcel=false;
  for(const fid of item.farms){
    const fs=farmStats(item,fid);
    if(fs.lastKg!=null) totalStock+=fs.lastKg;
    if(fs.usage!=null){ totalUsage+=fs.usage; hasUsage=true; if(fs.source==="excel") hasExcel=true; }
  }
  const pendingKg=(item.orders||[]).filter(o=>!o.received).reduce((s,o)=>s+o.kg,0);
  let suggestedOrder=null, monthsLeft=null;
  if(hasUsage&&totalUsage>0){
    const eff=totalStock+pendingKg; monthsLeft=eff/totalUsage;
    const need=totalUsage*(1+(item.reserveMonths||0));
    suggestedOrder=Math.max(0,Math.ceil(need-eff));
  }
  return { totalStock,pendingKg,totalUsage:hasUsage?totalUsage:null,monthsLeft,suggestedOrder,hasExcel };
}
function simulateMonth(item, scope, ymStr){
  const [Y,M]=ymStr.split("-").map(Number); const dim=daysInMonth(Y,M);
  let startDate,startKg,dailyUse; const orders=item.orders||[];
  if(scope==="all"){
    let total=0,usageSum=0,latest=null,ok=false;
    for(const fid of item.farms){ const fs=farmStats(item,fid);
      if(fs.lastKg!=null) total+=fs.lastKg;
      if(fs.lastDate&&(!latest||fs.lastDate>latest)) latest=fs.lastDate;
      if(fs.usage!=null){ usageSum+=fs.usage; ok=true; } }
    startDate=latest; startKg=total; dailyUse=ok?usageSum/30:0;
  } else { const fs=farmStats(item,scope); startDate=fs.lastDate; startKg=fs.lastKg||0; dailyUse=(fs.usage||0)/30; }
  if(!startDate) return null;
  let farmShare=1;
  if(scope!=="all"){ let us=0,mine=0; for(const fid of item.farms){ const u=farmStats(item,fid).usage||0; us+=u; if(fid===scope) mine=u; } farmShare=us>0?mine/us:1/item.farms.length; }
  const arrivals={}, arrInfo={};
  for(const o of orders){ const k=o.kg*(scope==="all"?1:farmShare);
    arrivals[o.etaDate]=(arrivals[o.etaDate]||0)+k;
    const c=arrInfo[o.etaDate]||{kg:0,anyPending:false,anyReceived:false}; c.kg+=k; if(o.received)c.anyReceived=true; else c.anyPending=true; arrInfo[o.etaDate]=c; }
  const cells=[]; let kg=startKg; const monthStart=iso(Y,M,1);
  let cursor=startDate;
  const stepTo=(t)=>{ while(cursor<t){ const d=new Date(cursor); d.setDate(d.getDate()+1); const nx=d.toISOString().slice(0,10); if(arrivals[nx]) kg+=arrivals[nx]; kg=Math.max(0,kg-dailyUse); cursor=nx; } };
  if(cursor<=monthStart) stepTo(monthStart);
  for(let day=1;day<=dim;day++){
    const date=iso(Y,M,day); const info=arrInfo[date];
    if(date<startDate){ cells.push({day,date,kg:null,before:true,info}); continue; }
    if(date===startDate){ cells.push({day,date,kg:startKg,anchor:true,info}); kg=startKg; cursor=date; continue; }
    if(arrivals[date]) kg+=arrivals[date]; kg=Math.max(0,kg-dailyUse); cursor=date;
    cells.push({day,date,kg,info});
  }
  return { cells,dailyUse,startDate,dim,Y,M };
}

  const addUnit=()=>setUnits(s=>[...s,{id:"u"+Date.now(),label:"",kg:""}]); const rmUnit=(i)=>setUnits(s=>s.filter((_,idx)=>idx!==i));

  const ok=name.trim()&&farms.length>0&&clean.length>0;
  return (<div style={S.fade}>
    <button style={S.back} onClick={onCancel}>← 戻る</button>
    <h2 style={S.h2}>餌を追加</h2>
    <label style={S.field}><span style={S.fieldLabel}>名称</span><input style={S.input} value={name} onChange={e=>setName(e.target.value)} placeholder="例: チモシー乾草"/></label>
    <div style={S.field}><span style={S.fieldLabel}>使用する農場</span><div style={S.farmRow}>{FARMS.map(x=><button key={x.id} style={{...S.farmTab,...(farms.includes(x.id)?S.farmActive:{})}} onClick={()=>toggleFarm(x.id)}>{x.name}</button>)}</div></div>
    <div style={S.field}><span style={S.fieldLabel}>単位パターン（個数で数える単位と1個あたりkg）</span>
      {units.map((u,i)=>(<div key={u.id} style={S.unitEditRow}>
        <input style={{...S.input,flex:1.4}} placeholder="ロール / コンテナ など" value={u.label} onChange={e=>setUnit(i,"label",e.target.value)}/>
        <input style={{...S.input,flex:1}} type="number" inputMode="decimal" placeholder="kg" value={u.kg} onChange={e=>setUnit(i,"kg",e.target.value)}/>
        {units.length>1&&<button style={S.rmBtn} onClick={()=>rmUnit(i)}>×</button>}</div>))}
      <button style={S.smallGhost} onClick={addUnit}>＋ パターンを追加</button></div>
    <label style={S.field}><span style={S.fieldLabel}>安全在庫（翌月棚卸しに加えて確保する月数）</span><input style={S.input} type="number" inputMode="decimal" value={reserve} onChange={e=>setReserve(e.target.value)}/></label>
    <button style={{...S.primaryBtn,opacity:ok?1:0.5}} disabled={!ok} onClick={()=>onSave({name:name.trim(),reserveMonths:parseFloat(reserve)||0,farms,units:clean})}>登録する</button>
  </div>);
}
